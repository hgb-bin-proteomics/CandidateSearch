97 public static int[] searchCPU(ref int[] candidatesValues, ref int[] candidatesIdx, ref int[] spectraValues, ref int[] spectraIdx,
98 int topN, float tolerance, bool normalize, bool useGaussianTol,
99 bool batched, int batchSize, bool useSparse, int cores, int verbose,
100 out int memStat)
101 {
102 var cValuesLoc = GCHandle.Alloc(candidatesValues, GCHandleType.Pinned);
103 var cIdxLoc = GCHandle.Alloc(candidatesIdx, GCHandleType.Pinned);
104 var sValuesLoc = GCHandle.Alloc(spectraValues, GCHandleType.Pinned);
105 var sIdxLoc = GCHandle.Alloc(spectraIdx, GCHandleType.Pinned);
106
107 int cVLength = candidatesValues.Length;
108 int cILength = candidatesIdx.Length;
109 int sVLength = spectraValues.Length;
110 int sILength = spectraIdx.Length;
111
112 var resultArray = new int[sILength * topN];
113
114 memStat = 1;
115
116 try
117 {
118 IntPtr cValuesPtr = cValuesLoc.AddrOfPinnedObject();
119 IntPtr cIdxPtr = cIdxLoc.AddrOfPinnedObject();
120 IntPtr sValuesPtr = sValuesLoc.AddrOfPinnedObject();
121 IntPtr sIdxPtr = sIdxLoc.AddrOfPinnedObject();
122
123 if (!batched && useSparse)
124 {
125 IntPtr result = findTopCandidates(cValuesPtr, cIdxPtr, sValuesPtr, sIdxPtr,
126 cVLength, cILength, sVLength, sILength,
127 topN, tolerance, normalize, useGaussianTol,
128 cores, verbose);
129
130 Marshal.Copy(result, resultArray, 0, sILength * topN);
131
132 memStat = releaseMemory(result);
133 }
134 else if (!batched && !useSparse)
135 {
136 IntPtr result = findTopCandidates2(cValuesPtr, cIdxPtr, sValuesPtr, sIdxPtr,
137 cVLength, cILength, sVLength, sILength,
138 topN, tolerance, normalize, useGaussianTol,
139 cores, verbose);
140
141 Marshal.Copy(result, resultArray, 0, sILength * topN);
142
143 memStat = releaseMemory(result);
144 }
145 else if (batched && useSparse)
146 {
147 IntPtr result = findTopCandidatesBatched(cValuesPtr, cIdxPtr, sValuesPtr, sIdxPtr,
148 cVLength, cILength, sVLength, sILength,
149 topN, tolerance, normalize, useGaussianTol, batchSize,
150 cores, verbose);
151
152 Marshal.Copy(result, resultArray, 0, sILength * topN);
153
154 memStat = releaseMemory(result);
155 }
156 else if (batched && !useSparse)
157 {
158 IntPtr result = findTopCandidatesBatched2(cValuesPtr, cIdxPtr, sValuesPtr, sIdxPtr,
159 cVLength, cILength, sVLength, sILength,
160 topN, tolerance, normalize, useGaussianTol, batchSize,
161 cores, verbose);
162
163 Marshal.Copy(result, resultArray, 0, sILength * topN);
164
165 memStat = releaseMemory(result);
166 }
167 else
168 {
169 Console.WriteLine("Impossible case!");
170 }
171 }
172 catch (Exception ex)
173 {
174 Console.WriteLine("Something went wrong:");
175 Console.WriteLine(ex.ToString());
176 memStat = 1;
177 }
178 finally
179 {
180 if (cValuesLoc.IsAllocated) { cValuesLoc.Free(); }
181 if (cIdxLoc.IsAllocated) { cIdxLoc.Free(); }
182 if (sValuesLoc.IsAllocated) { sValuesLoc.Free(); }
183 if (sIdxLoc.IsAllocated) { sIdxLoc.Free(); }
184 }
185
186 return resultArray;
187 }
